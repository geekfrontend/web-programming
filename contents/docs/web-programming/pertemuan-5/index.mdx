## 2) 🧩 Implementasi Bertahap

### Langkah 1 — Membuat util validasi form

**Tujuan**

- Memisahkan logic validasi agar komponen form tetap rapi.

**Buat file baru** `src/utils/noteValidation.ts`

```ts
export type NoteFormValues = {
  title: string;
  body: string;
};

export type NoteFormErrors = {
  title?: string;
  body?: string;
};

export function validateNoteForm(values: NoteFormValues): NoteFormErrors {
  const errors: NoteFormErrors = {};

  if (!values.title.trim()) {
    errors.title = "Judul wajib diisi";
  } else if (values.title.trim().length < 3) {
    errors.title = "Judul minimal 3 karakter";
  }

  if (values.body.trim().length > 250) {
    errors.body = "Isi catatan maksimal 250 karakter";
  }

  return errors;
}
```

**Verifikasi**

1. Simpan file dan cek tidak ada error.
2. Pastikan tipe `NoteFormErrors` dikenali saat di-import.

### Langkah 2 — Upgrade `NoteForm` dengan validasi

**Tujuan**

- Menampilkan error sebelum data dikirim ke reducer.

**Ubah file** `src/components/NoteForm.tsx`

```tsx
import { useState } from "react";
import { validateNoteForm, type NoteFormErrors } from "../utils/noteValidation";

type NoteFormProps = {
  onAddNote: (title: string, body: string) => void;
  submitting: boolean;
};

export function NoteForm({ onAddNote, submitting }: NoteFormProps) {
  const [title, setTitle] = useState("");
  const [body, setBody] = useState("");
  const [errors, setErrors] = useState<NoteFormErrors>({});

  function handleSubmit(event: React.FormEvent<HTMLFormElement>) {
    event.preventDefault();
    const nextErrors = validateNoteForm({ title, body });
    setErrors(nextErrors);
    if (Object.keys(nextErrors).length > 0) return;
    onAddNote(title, body);
    setTitle("");
    setBody("");
    setErrors({});
  }

  return (
    <section className="note-form">
      <h2>Catatan Baru</h2>
      <form onSubmit={handleSubmit}>
        <label>
          Judul
          <input
            type="text"
            value={title}
            onChange={(event) => setTitle(event.target.value)}
            placeholder="Contoh: Rangkuman kelas"
          />
        </label>
        {errors.title ? <p>{errors.title}</p> : null}
        <label>
          Isi
          <textarea
            value={body}
            onChange={(event) => setBody(event.target.value)}
            placeholder="Tulis isi catatan..."
          />
        </label>
        <p>{body.length}/250 karakter</p>
        {errors.body ? <p>{errors.body}</p> : null}
        <button type="submit" disabled={submitting}>
          {submitting ? "Menyimpan..." : "Simpan"}
        </button>
      </form>
    </section>
  );
}
```

**Verifikasi**

1. Submit judul kosong, pastikan pesan error muncul.
2. Isi body lebih dari 250 karakter, pastikan validasi aktif.
3. Isi data valid, pastikan catatan tetap bisa ditambahkan.

### Langkah 4 — Integrasi list rendering berbasis hasil filter

**Tujuan**

- Menampilkan daftar yang relevan berdasarkan input pencarian.

**Ubah file** `src/App.tsx`

```tsx
import { useMemo, useReducer, useState } from "react";
import { NoteHeader } from "./components/NoteHeader";
import { NoteForm } from "./components/NoteForm";
import { NoteList } from "./components/NoteList";
import { SearchBar } from "./components/SearchBar";
import { initialNotesState, noteReducer } from "./reducers/noteReducer";
import type { Note } from "./types/note";

export default function App() {
  const [state, dispatch] = useReducer(noteReducer, initialNotesState);
  const [query, setQuery] = useState("");

  function handleAddNote(title: string, body: string) {
    dispatch({ type: "add_start" });
    const nextNote: Note = {
      id: crypto.randomUUID(),
      title: title.trim(),
      body: body.trim(),
    };
    dispatch({ type: "add_success", payload: nextNote });
  }

  function handleDeleteNote(id: string) {
    dispatch({ type: "remove", payload: id });
  }

  const filteredNotes = useMemo(() => {
    const normalizedQuery = query.trim().toLowerCase();
    if (!normalizedQuery) return state.items;
    return state.items.filter((note) => {
      const title = note.title.toLowerCase();
      const body = note.body.toLowerCase();
      return title.includes(normalizedQuery) || body.includes(normalizedQuery);
    });
  }, [query, state.items]);

  return (
    <main className="page">
      <NoteHeader />
      <SearchBar query={query} onChangeQuery={setQuery} />
      <p>Total catatan: {state.items.length}</p>
      <p>Hasil ditampilkan: {filteredNotes.length}</p>
      <NoteForm onAddNote={handleAddNote} submitting={state.submitting} />
      <NoteList
        notes={filteredNotes}
        onDeleteNote={handleDeleteNote}
        emptyMessage={
          query.trim()
            ? "Catatan tidak ditemukan. Coba kata kunci lain."
            : "Belum ada catatan. Tambahkan catatan pertama kamu."
        }
      />
    </main>
  );
}
```

**Verifikasi**

1. Ketik kata kunci yang cocok, list terfilter sesuai input.
2. Ketik kata kunci yang tidak cocok, list kosong.
3. Counter total dan hasil tampil memperlihatkan perbedaan.

### Langkah 5 — Menyesuaikan `NoteList` untuk empty state dinamis

**Tujuan**

- Menjaga komponen list reusable untuk berbagai konteks kondisi kosong.

**Ubah file** `src/components/NoteList.tsx`

```tsx
import { NoteCard } from "./NoteCard";
import type { Note } from "../types/note";

type NoteListProps = {
  notes: Note[];
  onDeleteNote: (id: string) => void;
  emptyMessage: string;
};

export function NoteList({ notes, onDeleteNote, emptyMessage }: NoteListProps) {
  if (notes.length === 0) {
    return <p>{emptyMessage}</p>;
  }

  return (
    <section className="note-list">
      {notes.map((note) => (
        <NoteCard
          key={note.id}
          id={note.id}
          title={note.title}
          body={note.body}
          onDelete={onDeleteNote}
        />
      ))}
    </section>
  );
}
```

**Verifikasi**

1. Saat query kosong dan data habis, empty message default tampil.
2. Saat query tidak cocok, empty message khusus pencarian tampil.
